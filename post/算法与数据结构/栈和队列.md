## 比较含退格的字符串

[题目链接](https://leetcode-cn.com/problems/backspace-string-compare)

![](img/算法与数据结构/backspace-string-compare.jpg)

- 栈模拟
- 双指针，从末尾开始模拟

```cpp
class Solution {
public:
    bool backspaceCompare(string S, string T) {
        stack<char> s1;
        stack<char> s2;
        for (char &it : S) {
            if (it != '#') s1.push(it);
            else if (!s1.empty()) s1.pop();
        }

        for (char &it : T) {
            if (it != '#') s2.push(it);
            else if (!s2.empty()) s2.pop();
        }
        return s1 == s2;
    }
};
```

```cpp
class Solution {
public:
    bool backspaceCompare(string S, string T) {
        int i = S.size() - 1;
        int j = T.size() - 1;
        int cntS = 0;
        int cntT = 0;
        while (i >= 0 || j >= 0) {
            while (i >= 0) {
                if (S[i] == '#') cntS++, --i;
                else if (cntS) --cntS, --i;
                else break;
            }

            while (j >= 0) {
                if (T[j] == '#') cntT++, --j;
                else if (cntT) --cntT, --j;
                else break;
            }

            if (i >= 0 && j >= 0 && S[i] != T[j]) return false;
            if ((i < 0) != (j < 0)) return false;
            i--; j--;
        }
        return true;
    }
};
```
